# Find only movies that contain the phrase 'young man' (in that order)

GET movies/_search
{
  "query": {
    "match_phrase": {
      "overview": "young man"
    }
  }
}

{
  "mappings": {
    "show": {
      "properties": {
        "title": {
          "type": "text",
          "analyzer": "english"
        }
      }
    }
  }
}

PUT test/show/1
{
  "title": "The car is the money"
}

{
  "query": {

    }
  }
}

      "fields": ["title", "overview"]

  }
